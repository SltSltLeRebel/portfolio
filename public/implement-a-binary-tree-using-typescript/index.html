Implement a Binary Tree Using Typescript

23 Mar, 2021

Introduction

The purpose of this article is to implement a binary tree, using Typescript and the concept of “genericity”.

Genericity allows us to specify a type for our binary tree (number, string, …).

Implementation

Single Node

Defining the fields

class SingleNode<T> {
  private value: T;

  private left: SingleNode<T> | null;

  private right: SingleNode<T> | null;

  // ...
}

Defining the constructor

class SingleNode<T> {
  // ...

  constructor(
    value: T,

    left: SingleNode<T> | null = null,

    right: SingleNode<T> | null = null
  ) {
    this.value = value;

    this.left = left;

    this.right = right;
  }

  // ...
}

Defining the getters and setters

class SingleNode<T> {
  // ...

  getValue(): T {
    return this.value;
  }

  setValue(value: T): void {
    this.value = value;
  }

  getLeft(): SingleNode<T> | null {
    return this.left;
  }

  setLeft(left: SingleNode<T>): void {
    this.left = left;
  }

  getRight(): SingleNode<T> | null {
    return this.right;
  }

  setRight(right: SingleNode<T>): void {
    this.right = right;
  }

  // ...
}

Defining a method : getHeight()

class SingleNode<T> {
  // ...

  getHeight(): number {
    let node: SingleNode<T> = this;

    let leftHeight: number = 0;

    let rightHeight: number = 0;

    if (node.left !== null) {
      leftHeight = node.left.getHeight();
    }

    if (node.right !== null) {
      rightHeight = node.right.getHeight();
    }

    return (leftHeight > rightHeight ? leftHeight : rightHeight) + 1;
  }

  // ...
}

BinaryTree

class BinaryTree<T> {
  private root: SingleNode<T>;

  constructor(root: SingleNode<T>) {
    this.root = root;
  }

  getHeight(): number {
    return this.root.getHeight();
  }
}

Usage

// Defining nodes

const firstNode: SingleNode<number> = new SingleNode<number>(94);

const secondNode: SingleNode<number> = new SingleNode<number>(3.14, null, firstNode);

const thirdNode: SingleNode<number> = new SingleNode<number>(89, secondNode);

const fourthNode: SingleNode<number> = new SingleNode<number>(56, null, thirdNode);

const fifthNode: SingleNode<number> = new SingleNode<number>(72, fourthNode);

// Building tree

const tree: BinaryTree<number> = new BinaryTree<number>(firstNode);

// Log tree's height

console.log(tree.getHeight());
